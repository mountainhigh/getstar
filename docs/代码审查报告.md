    const res = await db.collection('children').doc(this.data.currentChildId).get();
    
    const child = res.data;
    
    // 计算本周积分
    const weekPoints = await this.calculatePeriodPoints(child._id, 'week');
    // 计算本月积分
    const monthPoints = await this.calculatePeriodPoints(child._id, 'month');
    
    this.setData({
      currentChild: {
        ...child,
        weekPoints,
        monthPoints
      }
    });
    
    // 计算等级进度
    this.calculateLevelProgress(child.points);
  } catch (err) {
    console.error('加载孩子信息失败:', err);
  }
}

async calculatePeriodPoints(childId, period) {
  const db = wx.cloud.database();
  const now = new Date();
  let startDate;
  
  if (period === 'week') {
    // 本周第一天
    const day = now.getDay() || 7;
    startDate = new Date(now);
    startDate.setDate(now.getDate() - day + 1);
    startDate.setHours(0, 0, 0, 0);
  } else if (period === 'month') {
    // 本月第一天
    startDate = new Date(now.getFullYear(), now.getMonth(), 1);
  }
  
  const res = await db.collection('check_ins').where({
    childId,
    createTime: db.command.gte(startDate)
  }).field({
    points: true
  }).get();
  
  return res.data.reduce((sum, item) => sum + item.points, 0);
}
